% The square
\newcommand\sq[2]{
  \fill[fill=cyan!30, draw=black, rounded corners, line width=1pt, shift={(\xxaxis:#1)},shift={(\yyaxis:#2)}] (0,0) -- (2,0) -- (2,-2) -- (0,-2) --cycle;
}

\newcommand\nsq[3]{
  \fill[fill=cyan!30, draw=black, rounded corners, line width=1pt, shift={(\xxaxis:#1)},shift={(\yyaxis:#2)}] (0,0) -- (2,0) -- (2,-2) -- (0,-2) --cycle;
\node at (#1+1,#2-1) {$#3$};
}

\newcommand\csq[2]{

}

}
